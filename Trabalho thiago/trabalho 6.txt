class Pilha {
  constructor() {
    this.acoes = [];
    this.historico = [];
  }

  empilharAcao(acao) {
    this.acoes.push(acao);
  }

  removerAcao() {
    if (this.isEmpty()) {
      return "A pilha está vazia";
    }

    return this.acoes.pop();
  }
  registroSaida() {
    this.historico.push(this.acoes.length);
  }

  quantidadeAcoes() {
    if (this.isEmpty()) {
      return "A pilha está vazia";
    }
    return this.acoes[this.acoes.length - 1];
  }

  isEmpty() {
    return this.acoes.length === 0;
  }

  acoesRealizadasNohistorico() {
    return this.acoes.length;
  }
}

let pilha = new Pilha();

pilha.empilharAcao(1);
pilha.empilharAcao(2);
pilha.empilharAcao(3);

console.log(registroSaida());
console.log(pilha.removerAcao());
console.log(pilha.quantidadeAcoes());
console.log(pilha.acoesRealizadasNohistorico());
