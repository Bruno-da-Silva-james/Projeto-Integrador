//Exercicio 4 Calculadora Básica

let cal1 = leitor.questionInt("Insira o primeiro numero: ");

let resultado

if (op === "+"){
 resultado = cal1 + cal2
}else if(op === "-"){
 resultado = cal1 - cal2
}else if(op === "*"){
 resultado = cal1 * cal2
}else if(op === "/"){
 resultado = cal1 / cal2
};
console.log(resultado);
